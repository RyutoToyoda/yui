// ==========================================
// 1. Firestore セキュリティルール (データベース用)
// ==========================================
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isSignedIn() {
      return request.auth != null;
    }

    function isJobOwner(jobId) {
      return isSignedIn() &&
        get(/databases/$(database)/documents/jobs/$(jobId)).data.creatorId == request.auth.uid;
    }

    function isValidEvaluation(value) {
      return value == null || value == "good" || value == "bad";
    }

    function isValidMutualEvaluation(data) {
      return isValidEvaluation(data.evaluation) &&
        isValidEvaluation(data.ownerEvaluation) &&
        isValidEvaluation(data.applicantEvaluation);
    }
    
    // ユーザー情報
    match /users/{userId} {
      allow read, write: if isSignedIn();
    }
    
    // 募集（ジョブ）
    match /jobs/{jobId} {
      allow read: if isSignedIn();
      allow create: if isSignedIn() && request.resource.data.creatorId == request.auth.uid;
      allow update, delete: if isJobOwner(jobId);
    }
    
    // 応募（アプリケーション）
    match /applications/{appId} {
      allow read: if isSignedIn();
      allow create: if isSignedIn() &&
        request.resource.data.applicantId == request.auth.uid &&
        isValidMutualEvaluation(request.resource.data);
      allow update: if isSignedIn() && (
        resource.data.applicantId == request.auth.uid ||
        (
          isJobOwner(resource.data.jobId) &&
          request.resource.data.applicantId == resource.data.applicantId &&
          request.resource.data.jobId == resource.data.jobId &&
          isValidMutualEvaluation(request.resource.data)
        )
      );
      allow delete: if isSignedIn() && (
        resource.data.applicantId == request.auth.uid ||
        isJobOwner(resource.data.jobId)
      );
    }
    
    // 決済履歴（トランザクション）
    match /transactions/{txnId} {
      allow read: if isSignedIn();
      allow create: if isSignedIn() && request.resource.data.fromUserId == request.auth.uid;
      allow update, delete: if false;
    }
    
    // スケジュール（スキマ時間）
    match /availabilities/{availId} {
      allow read: if isSignedIn();
      allow create: if isSignedIn() && request.resource.data.userId == request.auth.uid;
      allow update, delete: if isSignedIn() && resource.data.userId == request.auth.uid;
    }

    // 通知
    match /notifications/{notifId} {
      allow read: if isSignedIn() && request.auth.uid == resource.data.userId;
      allow create: if isSignedIn();
      allow update: if isSignedIn() && request.auth.uid == resource.data.userId;
      allow delete: if false;
    }

    // 広告（Ads）
    match /ads/{adId} {
      allow read: if true;
      allow write: if isSignedIn();
    }
  }
}

// ==========================================
// 2. Storage セキュリティルール (ファイル/画像用)
// ==========================================
rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    
    // 広告画像（adsフォルダ）
    match /ads/{allPaths=**} {
      allow read: if true; // 全員に公開
      allow write: if request.auth != null; // ログイン済みユーザーが管理
    }

    // その他すべてのファイル
    match /{allPaths=**} {
      allow read, write: if request.auth != null;
    }
  }
}
